// Task.cs - model danych public class TaskItem { [PrimaryKey, AutoIncrement] public int Id { get; set; } public string Name { get; set; } public bool IsCompleted { get; set; } public bool IsDeleted { get; set; } } // TaskDatabase.cs - obsługa bazy danych public class TaskDatabase { private readonly SQLiteAsyncConnection _database; public TaskDatabase(string dbPath) { _database = new SQLiteAsyncConnection(dbPath); _database.CreateTableAsync().Wait(); } public Task> GetTasksAsync() => _database.Table().Where(t => !t.IsDeleted).ToListAsync(); public Task> GetDeletedTasksAsync() => _database.Table().Where(t => t.IsDeleted).ToListAsync(); public Task SaveTaskAsync(TaskItem task) => task.Id != 0 ? _database.UpdateAsync(task) : _database.InsertAsync(task); public Task DeleteTaskAsync(TaskItem task) { task.IsDeleted = true; return _database.UpdateAsync(task); } public Task RestoreTaskAsync(TaskItem task) { task.IsDeleted = false; return _database.UpdateAsync(task); } } // MainPage.xaml
